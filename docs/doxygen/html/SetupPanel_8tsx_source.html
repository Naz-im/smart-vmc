54
55 const result = await check(permissionType);

64 if (!hasPermission) {

66 return;

69 Geolocation.getCurrentPosition(

73 },
74 (error) => {
75 console.log('Erreur GPS:', error.code, error.message);
76 },
77 { enableHighAccuracy: true, timeout: 15000, maximumAge: 10000 }
